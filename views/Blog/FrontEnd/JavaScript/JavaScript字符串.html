function phoneFormat(mobile){
	if(typeof mobile !== 'string'){
		return;
	} else {
		if(mobile.length === 11){
			return mobile.substr(0,4) + "*".repeat(4) + mobile.substr(8,3)
		} else {
			return
		}
	}
}
